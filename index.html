// 修改後的在線分析功能：確保 AI 讀得到清單資訊 async function fnAnalyze() { const otext = document.getElementById("text"); const resDisplay = document.getElementById("result-display"); // 1. 抓取輸入框內容 const currentInput = otext.value.trim(); // 2. 抓取留言區（清單）所有內容 const msgs = document.getElementsByClassName("msg-text"); let listContent = ""; for (let i = 0; i < msgs.length; i++) { listContent += `項目 ${i + 1}: ${msgs[i].innerText}\n`; } if (!currentInput && !listContent) return alert("清單與輸入框皆無內容可供分析"); resDisplay.style.display = "block"; resDisplay.innerText = "⏳ 正在提取 Maverick 知識清單核心（含留言區資訊）..."; // 3. 組合完整的 Prompt const fullPrompt = `你現在是 Maverick 的知識提純助理。 請針對以下【留言區清單】的所有內容，結合【當前輸入指令】，進行深度分析與提純。 【留言區清單內容】： ${listContent} 【當前輸入指令】： ${currentInput} 請直接輸出分析結果，並保持專業、精確的提純格式。`; try { const res = await fetch('/api/analyze', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: fullPrompt }] }] }) }); const data = await res.json(); if (data.candidates) { resDisplay.innerText = data.candidates[0].content.parts[0].text; } else { throw new Error(data.error || "分析失敗"); } } catch (e) { resDisplay.innerText = "❌ 錯誤：無法連線至分析通道，請檢查 Netlify 部署狀態。"; } }
